#!/usr/bin/sh

exit # don't run this script automatically - too risky

# to change to lts kernel:
pacman -S linux-lts
# check if kernel, ramdisk, and fallback are available:
ls -lA /boot
# remove standard kernel:
pacman -R linux
# UPDATE GRUB!!! DON'T FORGET THIS!!!
sudo grub-mkconfig -o /boot/grub/grub.cfg
